def print_df(df):
    with pd.option_context(
            'display.max_rows', None,
            'display.max_columns', None,
            'display.precision', 3,
        ):
        print(df)

def print_count_per_target(desc, df):
    count_per_target = df.groupby([target])[target].count()
    print(f'Count per {target} ({desc}):\n{count_per_target}\n')
    prop_per_target = count_per_target/count_per_target.sum()
    print(f'Proportion per {target} ({desc}):\n{prop_per_target}\n')
